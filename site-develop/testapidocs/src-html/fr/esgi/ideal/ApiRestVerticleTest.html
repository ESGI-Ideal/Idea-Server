003import io.vertx.core.DeploymentOptions;
004import io.vertx.core.Vertx;
005import io.vertx.core.VertxOptions;
006import io.vertx.core.json.JsonObject;
007import io.vertx.ext.unit.Async;
008import io.vertx.ext.unit.TestContext;
009import io.vertx.ext.unit.junit.VertxUnitRunner;
010import lombok.NonNull;

015
016import java.io.IOException;
017import java.net.ServerSocket;
018
019@RunWith(VertxUnitRunner.class)
020public class ApiRestVerticleTest {
021    private /*final static*/ int port = 8081;
022    private Vertx vertx;
023
024    //@Before
025    public void setUp(@NonNull final TestContext context) throws IOException {
026        this.vertx = Vertx.vertx(new VertxOptions().setBlockedThreadCheckInterval(200000000));
027        try(final ServerSocket socket = new ServerSocket(0)) {
028            this.port = socket.getLocalPort();
029        }
030        this.vertx.deployVerticle(ApiRestVerticle.class.getName(),
031                                  new DeploymentOptions().setConfig(new JsonObject().put("http.port", port)),
032                                  context.asyncAssertSuccess());
033    }
034
035    //@After
036    public void tearDown(@NonNull final TestContext context) {
037        this.vertx.close(context.asyncAssertSuccess());
038    }
039
040    /**
041     * Rigourous Test :-)
042     *
043     * @param context vertx's test context
044     */
045    //@Test
046    public void testApp(@NonNull final TestContext context) {
047        final Async async = context.async();
048        this.vertx.createHttpClient().getNow(port, "localhost", "/", resp -> resp.handler(body -> {
049            context.assertTrue(body.toString().contains("Hello World !"));
050            async.complete();
051        }));
052    }
