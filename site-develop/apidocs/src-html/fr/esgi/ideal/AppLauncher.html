001package fr.esgi.ideal;
002
003import io.vertx.core.DeploymentOptions;
004import io.vertx.core.Launcher;
005import io.vertx.core.Vertx;
006import io.vertx.core.VertxOptions;
007import io.vertx.core.VertxOptionsJson;
008import io.vertx.core.buffer.Buffer;
009import io.vertx.core.json.JsonObject;
010import io.vertx.core.logging.Logger;
011import io.vertx.core.logging.LoggerFactory;
012import io.vertx.core.logging.SLF4JLogDelegateFactory;
013import org.slf4j.bridge.SLF4JBridgeHandler;
014
015import java.io.IOException;
016import java.io.InputStream;
017
018import static io.vertx.core.logging.LoggerFactory.LOGGER_DELEGATE_FACTORY_CLASS_NAME;
019
020public class AppLauncher extends Launcher {
021    static {
022        System.setProperty("java.util.logging.config.file", "logging.properties");
023        System.setProperty(LOGGER_DELEGATE_FACTORY_CLASS_NAME, SLF4JLogDelegateFactory.class.getName());
024        //System.setProperty("vertx.logger-delegate-factory-class-name", "io.vertx.core.logging.SLF4JLogDelegateFactory");
025        //LoggerFactory.getLogger(LoggerFactory.class); // Required for Logback to work in Vertx
026    }
027    //static final Logger logger = LoggerFactory.getLogger("io.vertx.core.logging.SLF4JLogDelegateFactory");
028    static final Logger mainLogger = LoggerFactory.getLogger(io.vertx.core.logging.SLF4JLogDelegateFactory.class);
029
030    private static Logger logger = null;
031
032    /**
033     * Main entry point.
034     *
035     * @param args the user command line arguments.
036     */
037    public static void main(final String... args) {
038        new AppLauncher().dispatch(args);
039    }
040
041    /**
042     * Utility method to execute a specific command.
043     *
044     * @param cmd  the command
045     * @param args the arguments
046     */
047    public static void executeCommand(final String cmd, final String... args) {
048        new AppLauncher().execute(cmd, args);
049    }
050
051    /**
052     * Hook for sub-classes of {@link Launcher} after the config has been parsed.
053     *
054     * @param config the read config, empty if none are provided.
055     */
056    @Override
057    public void afterConfigParsed(final JsonObject config) {
058        logger = LoggerFactory.getLogger(AppLauncher.class);
059        logger.trace("afterConfigParsed : config={}", config.encodePrettily());
060        this.config = config;
061        logger.trace("Loading default config and override with custom config ...");
062        logger.trace("default config file at {}", this.getClass().getClassLoader().getResource("api-conf.json"));
063        try(final InputStream file = this.getClass().getClassLoader().getResourceAsStream("api-conf.json")) {
064            final JsonObject def_conf = new JsonObject(Buffer.buffer(Utils.getStringFromInputStream(file)));
065            config.mergeIn(def_conf.mergeIn(config, true), 1);
066        } catch(final IOException e) {
067            throw new RuntimeException(e);
068        }
069        super.afterConfigParsed(config);
070    }
071
072    private JsonObject config;
073
074    /**
075     * Hook for sub-classes of {@link Launcher} before the vertx instance is started.
076     *
077     * @param options the configured Vert.x options. Modify them to customize the Vert.x instance.
078     */
079    @Override
080    public void beforeStartingVertx(final VertxOptions options) {
081        System.setProperty(LOGGER_DELEGATE_FACTORY_CLASS_NAME, SLF4JLogDelegateFactory.class.getName());
082        SLF4JBridgeHandler.removeHandlersForRootLogger(); // Optionally remove existing handlers attached to j.u.l root logger
083        SLF4JBridgeHandler.install(); // add SLF4JBridgeHandler to j.u.l's root logger, should be done once during the initialization phase of your application
084        //InternalLoggerFactory.setDefaultFactory(Slf4JLoggerFactory.INSTANCE); // Force Netty to use Slf4j
085        this.lookups.forEach(cfl -> logger.trace("CommandFactoryLookup : {}", cfl));
086        logger.trace("beforeStartingVertx : options={}", options);
087        if(config.containsKey("VertxOptions")) {
088            VertxOptionsJson.fromJson(config.getJsonObject("VertxOptions"/*, new JsonObject()*/), options);
089            logger.debug("New VertxOptions : {}", options);
090        }
091        super.beforeStartingVertx(options);
092    }
093
094    /**
095     * Hook for sub-classes of {@link Launcher} after the vertx instance is started.
096     *
097     * @param vertx the created Vert.x instance
098     */
099    @Override
100    public void afterStartingVertx(final Vertx vertx) {
101        logger.trace("afterStartingVertx : vertx={}", vertx);
102        super.afterStartingVertx(vertx);
103    }
104
105    /**
106     * Hook for sub-classes of {@link Launcher} before the verticle is deployed.
107     *
108     * @param deploymentOptions the current deployment options. Modify them to customize the deployment.
109     */
110    @Override
111    public void beforeDeployingVerticle(final DeploymentOptions deploymentOptions) {
112        logger.trace("beforeDeployingVerticle : deploymentOptions={}", deploymentOptions);
113        super.beforeDeployingVerticle(deploymentOptions);
114    }
115
116    @Override
117    public void beforeStoppingVertx(final Vertx vertx) {
118        logger.trace("beforeStoppingVertx : vertx={}", vertx);
119        super.beforeStoppingVertx(vertx);
120    }
121
122    @Override
123    public void afterStoppingVertx() {
124        logger.trace("afterStoppingVertx");
125        super.afterStoppingVertx();
126    }
127
128    /**
129     * A deployment failure has been encountered. You can override this method to customize the behavior.
130     * By default it closes the `vertx` instance.
131     *
132     * @param vertx             the vert.x instance
133     * @param mainVerticle      the verticle
134     * @param deploymentOptions the verticle deployment options
135     * @param cause             the cause of the failure
136     */
137    @Override
138    public void handleDeployFailed(final Vertx vertx, final String mainVerticle, final DeploymentOptions deploymentOptions, final Throwable cause) {
139        logger.warn("handleDeployFailed : mainVerticle={} ; deploymentOptions={} ; cause={}", mainVerticle, deploymentOptions, cause);
140        super.handleDeployFailed(vertx, mainVerticle, deploymentOptions, cause);
141    }
142}
