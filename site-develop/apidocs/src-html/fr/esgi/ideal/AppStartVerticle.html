001package fr.esgi.ideal;
002
003import io.vertx.core.AbstractVerticle;
004import io.vertx.core.CompositeFuture;
005import io.vertx.core.DeploymentOptions;
006import io.vertx.core.DeploymentOptionsJson;
007import io.vertx.core.Future;
008import io.vertx.core.json.JsonObject;
009import lombok.NonNull;
010import lombok.extern.slf4j.Slf4j;
011
012import java.lang.management.ManagementFactory;
013import java.util.stream.Collectors;
014
015@Slf4j
016public final class AppStartVerticle extends AbstractVerticle {
017    @Override
018    public void start(final Future<Void> startFuture) {
019        log.trace("process = {}", ManagementFactory.getRuntimeMXBean().getName());
020        log.debug("config() = {}", this.config().encodePrettily());
021        CompositeFuture.join(
022            this.config().getJsonObject("verticles", new JsonObject()).stream()
023                .peek(e -> log.debug("found from config verticle {}", e.getKey()))
024                .map(e -> this.deploy(e.getKey(), DeploymentOptionsJson.fromJson(((JsonObject) e.getValue()).getJsonObject("deploymentOptions", new JsonObject()), new DeploymentOptions())))
025                .collect(Collectors.toList())
026        ).setHandler(ar -> {
027            if(ar.succeeded())
028                startFuture.complete();
029            else
030                startFuture.fail(ar.cause());
031        });
032        log.info("Module(s) and/or verticle(s) deployment... DONE");
033    }
034
035    @Override
036    public void stop(final Future<Void> stopFuture) {
037        log.debug("Undeploying verticle(s)... DONE");
038        this.vertx.close();
039        log.info("Application stopped successfully.");
040        stopFuture.complete();
041    }
042
043    private Future<Void> deploy(@NonNull final String clazz, @NonNull final DeploymentOptions options) {
044        final Future<Void> future = Future.future();
045        log.trace("while deploy {} with {}", clazz, options.toJson().encodePrettily());
046        this.vertx.deployVerticle(clazz, options, handler -> {
047            if(handler.succeeded()) {
048                log.info("{} started successfully (deployment identifier: {})", clazz, handler.result());
049                future.complete();
050            } else {
051                log.error(clazz+" deployment failed due to: ", handler.cause());
052                future.fail(handler.cause());
053            }
054        });
055        return future;
056    }
057}
