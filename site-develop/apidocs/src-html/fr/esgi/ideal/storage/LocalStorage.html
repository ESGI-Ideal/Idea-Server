025    @Override
026    public void init(@NonNull final Vertx vertx, @NonNull final JsonObject configStorage, @NonNull final Future<Void> future) {
027        this.vertx = vertx;
028        this.assets = Paths.get(configStorage.getString("assets_dir", "./ideal-assets"));
029        //
030        this.vertx.fileSystem().exists(this.assets.toAbsolutePath().toString(), res -> {
031            if(res.succeeded()) {
032                if(!res.result())
033                    this.vertx.fileSystem().mkdirs(this.assets.toAbsolutePath().toString(), res2 -> {
034                       if(res2.succeeded()) {
035                           if(!Files.isDirectory(this.assets))
036                               future.fail(new InvalidPathException(this.assets.toString(), "The assets path is not an directory"));
037                           else if(!Files.isReadable(this.assets) || !Files.isWritable(this.assets))
038                               future.fail(new InvalidPathException(this.assets.toString(), "Haven't right acces to assets directory"));
039                           else
040                               future.complete();
041                       } else
042                           future.fail(res2.cause());
043                    });
044                else
045                    future.complete();
046            } else
047                future.fail(res.cause());
048        });
049    }
050
051    @Override

054    }

094    @Override
095    public Future<Void> delete(@NonNull final TypeObject type, @NonNull final Long id) {
096        final Future<Void> future = Future.future();
097        vertx.fileSystem().delete(generatePath(type, id), future.completer());
098        return future;
099    }
100
101    @Override
102    public Future<Void> upload(@NonNull final Long id, @NonNull final TypeObject type, @NonNull final String contentType, @NonNull final ReadStream<Buffer> asyncInput) {
103        final Future<Void> future = Future.future();
104        //this.vertx.fileSystem().writeFile(generatePath(type, id), asyncInput, future.completer());
105        asyncInput.pause();
106        this.vertx.fileSystem().open(generatePath(type, id), new OpenOptions(), res -> {
107            final Pump pump = Pump.pump(asyncInput, res.result());
108            asyncInput.endHandler(end -> res.result().close(future.completer()));
109            pump.start();
110            asyncInput.resume();
111        });
112        return future;
113    }
