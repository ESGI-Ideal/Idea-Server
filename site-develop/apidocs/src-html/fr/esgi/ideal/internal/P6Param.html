075    P6Param(@NonNull final String base, @NonNull final String driver,
076                    final String ds_basic, final String ds_xa, final String ds_poll,
077                    final String dateFormat) {
078        this.URI_base = base;

080        this.basicDS = Optional.ofNullable(ds_basic);
081        this.xaDS = Optional.ofNullable(ds_xa);
082        this.pollingDS = Optional.ofNullable(ds_poll);
083        this.dateFormat = Optional.ofNullable(dateFormat);
084    }

086    P6Param(@NonNull final String base, @NonNull final String driver,
087                    final String ds_basic, final String ds_xa, final String ds_poll) {

089    }
090
091    P6Param(@NonNull final String base, @NonNull final String driver) {
