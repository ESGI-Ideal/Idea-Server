023/**
024 * Utility class for vertx's incapability to read from jar ...

026@Slf4j

028public class FSIO {
029    /**
030     *
031     * @param resourcePath the intern file to copy
032     * @return {@link Path} of file copied
033     * @throws IOException
034     */
035    public static Path getResourceAsExternal(@NonNull final String resourcePath) throws IOException {
036        try(final InputStream in = FSIO.class.getClassLoader().getResourceAsStream(resourcePath)) {
037            final Path file = newTmpFile();
038            log.debug("copy resource {} in {}", resourcePath, file);
039            Files.copy(in, file, StandardCopyOption.REPLACE_EXISTING);
040            return file;
041        }
042    }
043
044    /**
045     *
046     * @param resourcesPath the intern files (should not start with "/" !)
047     * @return the {@link Path} of folder with files
048     * @throws IOException
049     */
050    public static Path getResourcesAsExternal(@NonNull final String... resourcesPath) throws IOException {
051        final Path tmpRoot = newTmpFolder();
052        for(@NonNull final String res : resourcesPath) {
053            try(final InputStream in = FSIO.class.getClassLoader().getResourceAsStream(res)) {
054                final Path file = tmpRoot.resolve(res);
055                log.debug("copy resource {} in {}", res, file);
056                Files.copy(in, file, StandardCopyOption.REPLACE_EXISTING);
057            }
058        }
059        return tmpRoot;
060    }
061
062    /**
063     *
064     * @param resourcesPath files to concat
065     * @return files concatenated
066     * @throws IOException
067     */
068    public static Path getResourcesAsExternalMerged(@NonNull final String... resourcesPath) throws IOException {
069        try(final InputStream in = new SequenceInputStream(Collections.enumeration(Arrays.stream(resourcesPath).sequential().map(FSIO.class.getClassLoader()::getResourceAsStream).collect(Collectors.toList())))) {
070            final Path file = newTmpFile();
071            log.debug("copy in {} resources {}", file, Arrays.toString(resourcesPath));
072            Files.copy(in, file, StandardCopyOption.REPLACE_EXISTING);
073            return file;
074        }
075    }
076
077    public static Path getResourcesYamlsMergedAsExternal(@NonNull final String... resourcesPath) throws IOException {
078        try {
079            //final ObjectMapper mapper = new ObjectMapper(new YAMLFactory());
080            final ObjectMapper mapper = io.swagger.v3.parser.ObjectMapperFactory.createYaml();
081            final Path file = newTmpFile();
082            log.debug("copy in {} resources {}", file, Arrays.toString(resourcesPath));
083            mapper.writeTree(new YAMLFactory().createGenerator(file.toFile(), JsonEncoding.UTF8), Arrays.stream(resourcesPath).map(res -> {
084                try (final InputStream in = FSIO.class.getClassLoader().getResourceAsStream(res)) {
085                    return mapper.readTree(in);
086                } catch (final IOException e) {
087                    throw new RuntimeException(e);
088                }
089            }).reduce((prev, next) -> {merge((ObjectNode)prev, (ObjectNode)next); return prev;}).orElseThrow(NoSuchElementException::new));
090            //new JsonObject((Map)Json.mapper.convertValue(new User(), Map.class));
091            return file;
092        } finally{}
093    }
094    //from https://github.com/addthis/codec/blob/master/src/main/java/com/addthis/codec/jackson/Jackson.java
095    private static void merge(final ObjectNode primary, final ObjectNode backup) {
096        backup.fieldNames().forEachRemaining(fieldName -> {
097            final JsonNode primaryValue = primary.get(fieldName);
098            if (primaryValue == null) {
099                primary.set(fieldName, backup.get(fieldName).deepCopy());
100            } else if(primaryValue.isObject()) {
101                final JsonNode backupValue = backup.get(fieldName);
102                if(backupValue.isObject())
103                    merge((ObjectNode) primaryValue, backupValue.deepCopy());
104            }
105        });
106    }

112     */
113    public static Path newTmpFile() throws IOException {
114        final Path file = Files.createTempFile("ideal-", null);
115        log.info("New temp file : {}", file);
116        //file.toFile().deleteOnExit();
117        return file;
118    }
119
120    /**
121     * Create a new temporary file
122     * @return
123     * @throws IOException
124     */
125    public static Path newTmpFolder() throws IOException {
126        final Path folder = Files.createTempDirectory("ideal-");
127        log.info("New temp folder : {}", folder);
128        folder.toFile().deleteOnExit();
129        return folder;
130    }
