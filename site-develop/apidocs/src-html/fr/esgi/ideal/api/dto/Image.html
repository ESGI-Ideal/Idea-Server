001package fr.esgi.ideal.api.dto;
002
003import lombok.AllArgsConstructor;
004import lombok.Builder;
005import lombok.Data;
006
007import javax.validation.constraints.Min;
008
009/**
010 * Information/metadata about an image
011 */
012@Data
013@Builder
014@AllArgsConstructor
015public class Image {
016    private Long id;
017
018    private byte[] hash;
019
020    private String filename;
021
022    private byte[] hashfile;
023
024    /**
025     * Size of the image
026     */
027    @Min(0)
028    private int height, width;
029}
