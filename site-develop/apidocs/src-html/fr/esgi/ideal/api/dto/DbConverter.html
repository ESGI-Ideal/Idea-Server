024/**
025 * Converter of beans between DB DAO and API DTO
026 */
027@UtilityClass
028public class DbConverter {
029    public static User toAPI(final IUsers db) {
030        if(db != null) {
031            final User user = User.builder()
032                    .id(db.getId())
033                    .inscription(db.getCreated())
034                    .isAdmin(db.getAdmin())
035                    .mail(db.getMail())
036                    .img(db.getImage()) //TODO
037                    .build();
038            return user;
039        } else
040            return null;
041    }
042
043    public static User jsonUser(final JsonObject db) {
044        if(db != null) {
045            final User user = User.builder()
046                    .id(db.getLong("id"))
047                    .inscription((OffsetDateTime)db.getValue("created"))
048                    .isAdmin(db.getBoolean("isAdmin", false))
049                    .mail(db.getString("email"))
050                    .img(db.getLong("img")) //TODO
051                    .build();
052            return user;
053        } else
054            return null;
055    }
056
057    public static Users toDB(final User api) {
058        if(api != null) {
059            final Users user = new Users(api.getId(), api.getImg(), api.getMail(), api.getInscription(), api.isAdmin());
060            return user;
061        } else
062            return null;
063    }
064
065    public static Article toAPI(final IArticles db) {
066        if(db != null) {
067            final Article article = Article.builder()
068                    .id(db.getId())
069                    .name(db.getName())
070                    .created(db.getCreated())
071                    .updated(db.getUpdated())
072                    .description(db.getDescription())
073                    .price(Optional.ofNullable(db.getPrice()).map(Double::floatValue).orElse(null))
074                    .customerRating(Math.toIntExact(db.getRate()))
075                    .img(db.getImage()) //TODO
076                    .build();
077            return article;
078        } else
079            return null;
080    }
081
082    public static Article jsonArticle(final JsonObject db) {
083        if(db != null) {
084            final Article article = Article.builder()
085                    .id(db.getLong("id"))
086                    .name(db.getString("name"))
087                    //.created(Optional.ofNullable((LocalDateTime)db.getValue("created")).map(DbConverter::toTimeApi)).orElse(null))
088                    .created((OffsetDateTime)db.getValue("created"))
089                    .updated((OffsetDateTime)db.getValue("updated"))
090                    .description(db.getString("description"))
091                    .price(db.getFloat("price"))
092                    .customerRating(db.getInteger("rate", null))
093                    .img(db.getLong("id")) //TODO
094                    .build();
095            return article;
096        } else
097            return null;
098    }
099
100    public static Articles toDB(final Article api) {
101        if(api != null) {
102            final Articles article = new Articles(api.getId(), api.getName(), api.getImg(), api.getDescription(), Optional.ofNullable(api.getPrice()).map(Float::doubleValue).orElse(null),
103                    api.getCreated(), api.getUpdated(), Optional.ofNullable(api.getCustomerRating()).map(Integer::longValue).orElse(null));
104            return article;
105        } else
106            return null;
107    }
108
109    public static Partner toAPI(final IPartners db) {
110        if(db != null) {
111            return Partner.builder()
112                    .id(db.getId())
113                    .name(db.getName())
114                    .description(db.getDescription())
115                    .img(db.getImage()) //TODO
116                    .build();
117        } else
118            return null;
119    }
120
121    public static Partner jsonPartner(final JsonObject db) {
122        if(db != null) {
123            return Partner.builder()
124                    .id(db.getLong("id"))
125                    .name(db.getString("name"))
126                    .description(db.getString("description"))
127                    .img(db.getLong("img")) //TODO
128                    .build();
129        } else
130            return null;
131    }
132
133    public static Partners toDB(final Partner api) {
134        if(api != null) {
135            final Partners partener = new Partners(api.getId(),api.getName(), api.getDescription(), api.getImg());
136            return partener;
137        } else
138            return null;
139    }
140
141    public static Ad toAPI(final IAds db) {
142        if(db != null) {
143            return Ad.builder()
144                    .id(db.getId())
145                    .description(db.getTitle())
146                    .img(db.getImage()) //TODO
147                    .build();
148        } else
149            return null;
150    }
151
152    public static Ad jsonAds(final JsonObject db) {
153        if(db != null) {
154            return Ad.builder()
155                    .id(db.getLong("id"))
156                    .description(db.getString("description"))
157                    .img(db.getLong("img")) //TODO
158                    .build();
159        } else
160            return null;
161    }
162
163    public static Ads toDB(final Ad api) {
164        if(api != null) {
165            final Ads ad = new Ads(api.getId(), api.getDescription(), api.getImg());
166            return ad;
167        } else
168            return null;
169    }
170
171    public static Image toAPI(final IImages db) {
172        if(db != null) {
173            return Image.builder()
174                    .id(db.getId())
175                    .height(db.getHeight())
176                    .width(db.getWidth())
177                    .filename(db.getFilename())
178                    .hash(db.getHash())
179                    .hashfile(db.getHashfile())
180                    .build();
181        } else
182            return null;
183    }
184
185    public static Images toDB(final Image api) {
186        if(api != null) {
187            final Images ad = new Images(api.getId(), api.getHash(), api.getFilename(), api.getHashfile(), api.getWidth(), api.getHeight());
188            return ad;
189        } else
190            return null;
191    }
192
193    private static LocalDateTime convert_(final Date date) {
194        if(date != null) {
195            return date.toInstant().atZone(ZoneId.systemDefault()).toLocalDateTime();
196            //return Instant.ofEpochMilli(date.getTime()).atZone(ZoneId.systemDefault()).toLocalDateTime();
197        } else
198            return null;
199    }
200
201    private static OffsetDateTime toTimeApi(@NonNull final LocalDateTime dateTime) {
202        return dateTime.atOffset(ZoneOffset.UTC);
203    }
204
205    private static LocalDateTime toTimeDB(@NonNull final OffsetDateTime dateTime) {
206        return dateTime.toLocalDateTime();
207    }
208}
