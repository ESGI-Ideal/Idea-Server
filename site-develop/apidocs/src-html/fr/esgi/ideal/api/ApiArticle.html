107    public void searchArticle(@NonNull final RoutingContext routingContext) {
108        final RequestParameters reqParams = routingContext.get("parsedParameters");
109        final int limit = Optional.ofNullable(reqParams.queryParameter("limit")).map(RequestParameter::getInteger).orElse(20);
110        final int offset = Optional.ofNullable(reqParams.queryParameter("offset")).map(RequestParameter::getInteger).orElse(20);
111        final Optional<String> orderBy = Optional.ofNullable(reqParams.queryParameter("orderby")).map(RequestParameter::getString)
112                                                 .map(String::trim)
113                                                 .map(String::toLowerCase);
114        final boolean order = "asc".equalsIgnoreCase(Optional.ofNullable(reqParams.queryParameter("order")).map(RequestParameter::getString).orElse("asc"));
115        final Optional<Set<String>> keywords = Optional.ofNullable(reqParams.queryParameter("query")).map(RequestParameter::getString)
116                .map(s -> Stream.of(s.split("\\s")).map(String::trim)
117                                                         .map(String::toLowerCase)
118                                                         .filter(kw -> !kw.isEmpty())
119                                                         .collect(Collectors.toSet()))
120                .filter(s -> !s.isEmpty());
121        if(keywords.isPresent()) {
122            final JsonObject jobj = new JsonObject().put("keywords", new JsonArray(new ArrayList<>(keywords.get())))
123                                                    .put("limit", limit)
124                                                    .put("offset", offset)
125                                                    .mergeIn(orderBy.map(ob -> new JsonObject().put("orderby", orderBy.get()).put("order", order))
126                                                                    .orElseGet(JsonObject::new));
127            this.eventBus.<Integer>send(DatabaseVerticle.DB_ARTICLE_SEARCH_TOTAL, jobj, asyncTotal -> {
128                if(asyncTotal.succeeded()) {
129                    this.eventBus.<Collection<Articles>>send(DatabaseVerticle.DB_ARTICLE_SEARCH, jobj, asyncMsg -> {
130                        if(asyncMsg.succeeded())
131                            RouteUtils.send(routingContext, HttpResponseStatus.OK, jobj.put("totalResult", asyncTotal.result().body())
132                                                                                       .put("result", asyncMsg.result().body()));
133                        else
134                            RouteUtils.error(routingContext, "An error occur on the server (phase 2)");
135                    });
136                } else
137                    RouteUtils.error(routingContext, "An error occur on the server (phase 1)");
138            });
139        } else
140            RouteUtils.error(routingContext, "The Id passed is null");
141    }

143    public void voteArticle(@NonNull final RoutingContext routingContext) {
144        final Optional<Long> id = Optional.ofNullable(((RequestParameters) routingContext.get("parsedParameters")).pathParameter("id"))
145                .map(RequestParameter::getLong);
146        final Optional<Boolean> value = Optional.ofNullable(((RequestParameters) routingContext.get("parsedParameters")).body())
147                .map(RequestParameter::getBoolean);
148        if(id.isPresent() && value.isPresent())
149            this.eventBus.<Void>send(DatabaseVerticle.DB_ARTICLE_VOTE,
150                                     new JsonObject().put("articleId", id).put("userId", routingContext.user().principal().getLong("id")).put("like", value),
151                                     asyncMsg -> {
152                if(asyncMsg.succeeded())
153                    RouteUtils.send(routingContext, HttpResponseStatus.OK, null);
154                else {
155                    log.error("Get error from bus resquest", asyncMsg.cause());
156                    RouteUtils.error(routingContext, "An error occur on the server");
157                }
158            });
159        else
160            RouteUtils.error(routingContext, "The Id passed is null");
161    }
162
163    public void unvoteArticle(@NonNull final RoutingContext routingContext) {
164        final Optional<Long> id = Optional.ofNullable(((RequestParameters) routingContext.get("parsedParameters")).pathParameter("id"))
165                                          .map(RequestParameter::getLong);
166        if(id.isPresent())
167            this.eventBus.<Void>send(DatabaseVerticle.DB_ARTICLE_UNVOTE, new JsonObject().put("articleId", id).put("userId", routingContext.user().principal().getLong("id")), asyncMsg -> {
168                if(asyncMsg.succeeded())
169                    RouteUtils.send(routingContext, HttpResponseStatus.OK, null);
170                else {
171                    log.error("Get error from bus resquest", asyncMsg.cause());
172                    RouteUtils.error(routingContext, "An error occur on the server");
173                }
174            });
175        else
176            RouteUtils.error(routingContext, "The Id passed is null");
177    }
