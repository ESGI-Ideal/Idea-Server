001package fr.esgi.ideal.api;
002
003import fr.esgi.ideal.DatabaseVerticle;
004import fr.esgi.ideal.api.dto.Article;
005import fr.esgi.ideal.api.dto.DbConverter;
006import fr.esgi.ideal.dao.tables.pojos.Articles;
007import io.vertx.core.Future;
008import io.vertx.core.eventbus.EventBus;
009import io.vertx.core.json.JsonObject;
010import lombok.AllArgsConstructor;
011import lombok.NonNull;
012import lombok.extern.slf4j.Slf4j;
013
014import java.util.List;
015import java.util.Optional;
016
017@Slf4j
018@AllArgsConstructor
019public class ApiArticle implements SubApiAlter<Articles, Article> {
020    private final EventBus eventBus;
021
022    @Override
023    public Article mapTo(final Articles obj) {
024        return DbConverter.toAPI(obj);
025    }
026
027    @Override
028    public Articles mapFrom(final Article obj) {
029        return DbConverter.toDB(obj);
030    }
031
032    @Override
033    public Article mapFrom(final JsonObject obj) {
034        return DbConverter.jsonArticle(obj);
035    }
036
037    @Override
038    public Future<List<Articles>> getAll() {
039        final Future<List<Articles>> future = Future.future();
040        this.eventBus.<List<Articles>>send(DatabaseVerticle.DB_ARTICLE_GET_ALL, null, asyncMsg -> {
041            if(asyncMsg.succeeded())
042                future.complete(asyncMsg.result().body());
043            else {
044                log.error("Get error from bus resquest", asyncMsg.cause());
045                future.fail(asyncMsg.cause());
046            }
047        });
048        return future;
049    }
050
051    @Override
052    public Future<Optional<Articles>> get(@NonNull final Long id) {
053        final Future<Optional<Articles>> future = Future.future();
054        this.eventBus.<Articles>send(DatabaseVerticle.DB_ARTICLE_GET_BY_ID, id, asyncMsg -> {
055            if(asyncMsg.succeeded())
056                future.complete(Optional.ofNullable(asyncMsg.result().body()));
057            else {
058                log.error("Get error from bus resquest", asyncMsg.cause());
059                future.fail(asyncMsg.cause());
060            }
061        });
062        return future;
063    }
064
065    @Override
066    public Future<Void> delete(@NonNull final Long id) {
067        final Future<Void> future = Future.future();
068        this.eventBus.<Void>send(DatabaseVerticle.DB_ARTICLE_DELETE_BY_ID, id, asyncMsg -> {
069            if(asyncMsg.succeeded())
070                future.complete();
071            else {
072                log.error("Get error from bus resquest", asyncMsg.cause());
073                future.fail(asyncMsg.cause());
074            }
075        });
076        return future;
077    }
078
079    @Override
080    public Future<Long> createOrUpdate(@NonNull final Article data) {
081        final Future<Long> future = Future.future();
082        this.eventBus.<Long>send(DatabaseVerticle.DB_ARTICLE_CREATE, DbConverter.toDB(data), asyncMsg -> {
083            if(asyncMsg.succeeded()) {
084                //future.complete(DbConverter.toAPI(asyncMsg.result().body()));
085                log.debug("Insert id {}", asyncMsg.result().body());
086                future.complete(asyncMsg.result().body()); }
087            else {
088                log.error("Get error from bus resquest", asyncMsg.cause());
089                future.fail(asyncMsg.cause());
090            }
091        });
092        return future;
093    }
