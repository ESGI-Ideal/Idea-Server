047    public void token(final RoutingContext context) {
048        try {
049            final String clientID = Optional.ofNullable(context.request().getParam("client_id")).orElseThrow(NullPointerException::new);
050            final String clientSecret = Optional.ofNullable(context.request().getParam("client_id")).orElseThrow(NullPointerException::new);
051            final String userName = Optional.ofNullable(context.request().getParam("username")).orElseThrow(NullPointerException::new);
052            final String userPsw = Optional.ofNullable(context.request().getParam("password")).orElseThrow(NullPointerException::new);
053            final String grantType = Optional.ofNullable(context.request().getParam("grant_type")).orElseThrow(NullPointerException::new);
054            final String[] scopes = Optional.ofNullable(context.request().getParam("scope")).orElseThrow(NullPointerException::new).split("\\s+");
055
056            final Logger logger = LoggerFactory.getLogger(this.getClass());
057            logger.info("type="+ grantType);
058            logger.info("name="+ userName);
059            logger.info("psw="+ userPsw);
060            logger.info("id="+ clientID);
061            logger.info("sct="+ clientSecret);
062            logger.info("scopes='"+ Arrays.toString(scopes) +"'");
063
064            switch(grantType) {
065                case "password":
066                    /*if(!this.accounts.containsKey(userName)) {
067                        context.response().setStatusCode(400).setStatusMessage("Unknown user").end();
068                        return;
069                    }
070                    final User user = this.accounts.get(userName);
071                    if(!Optional.ofNullable(user.getPassword()).orElse("").equals(userPsw)) {
072                        context.response().setStatusCode(400).setStatusMessage("Invalid credentials").end();
073                        return;
074                    }*/
075                    //final String accessToken = tokenFountain.nextAccessToken();
076                    JsonObject response = new JsonObject()
077                            .put("access_token", Base64.getEncoder().encodeToString((userName+" "+new Date().getTime()).getBytes(StandardCharsets.UTF_8)))
078                            .put("token_type", "bearer")
079                            .put("expires_in", 3600)
080                            //.put("refresh_token", "")
081                            .put("scope", Arrays.stream(scopes).collect(Collectors.joining(" ")));
082                    context.response().putHeader("Cache-Control", "no-store").putHeader("Pragma", "no-cache")
083                            .putHeader("Content-Type", "application/json")
084                            .end(response.encodePrettily());
085                    break;
086                /*case "refresh_token": //TODO ?
087                    break;*/
088                default:
089                    //log.error
090                    context.response().setStatusCode(400).setStatusMessage("unsupported_grant_type").end("grant_type " + grantType + " must be password");
091                    break;
092            }
093        } catch(final NullPointerException n) {
094            context.response().setStatusCode(400).setStatusMessage("missing argument").end();
095        }
096    }

102    public void prepare_oauth(final RoutingContext routingContext) {
103        final String raw = routingContext.request().getHeader("Authorization");
104        if((raw != null) && raw.startsWith("Bearer ")) {
105            final String tokenDecode = new String(Base64.getDecoder().decode(raw.split(" ")[1]));
106            final String[] tokenRaw = tokenDecode.split(" ");
107            logger.info("token: "+Arrays.toString(tokenRaw));
108            //routingContext.setUser(this.accounts.get(tokenRaw[0]));
109            routingContext.setUser(tmp);
110            //TODO routingContext.data().put(KEY_TOKEN_LIMIT, new Date(tokenRaw[1]));
111        }
112        routingContext.next();
113    }

128    public void check_scope_user(final RoutingContext routingContext) {
129        if(verif_token_limit(routingContext))
130            routingContext.user().isAuthorized("user", res -> {
131                if(res.succeeded() && res.result())
132                    routingContext.next();
133                else
134                    routingContext.fail(404);
135            });
136    }
