001package fr.esgi.ideal.api;
002
003import fr.esgi.ideal.DatabaseVerticle;
004import fr.esgi.ideal.api.dto.Ad;
005import fr.esgi.ideal.api.dto.DbConverter;
006import fr.esgi.ideal.dao.tables.pojos.Ads;
007import io.vertx.core.Future;
008import io.vertx.core.eventbus.EventBus;
009import io.vertx.core.json.Json;
010import io.vertx.core.json.JsonObject;
011import io.vertx.ext.web.RoutingContext;
012import io.vertx.ext.web.api.RequestParameters;
013import lombok.AllArgsConstructor;
014import lombok.NonNull;
015import lombok.extern.slf4j.Slf4j;
016
017import java.util.List;
018import java.util.Optional;
019import java.util.concurrent.atomic.AtomicLong;
020
021@Slf4j
022@AllArgsConstructor
023public class ApiAd implements SubApiAlter<Ads, Ad> {
024    private final EventBus eventBus;
025    private final static AtomicLong rotate = new AtomicLong(0L);
026
027    @Override
028    public Ad mapTo(final Ads obj) {
029        return DbConverter.toAPI(obj);
030    }
031
032    @Override
033    public Ads mapFrom(final Ad obj) {
034        return DbConverter.toDB(obj);
035    }
036
037    @Override
038    public Ad mapFrom(final JsonObject obj) {
039        return DbConverter.jsonAds(obj);
040    }
041
042    @Override
043    public Future<List<Ads>> getAll() {
044        final Future<List<Ads>> future = Future.future();
045        this.eventBus.<List<Ads>>send(DatabaseVerticle.DB_AD_GET_ALL, null, asyncMsg -> {
046            if(asyncMsg.succeeded())
047                future.complete(asyncMsg.result().body());
048            else {
049                log.error("Get error from bus resquest", asyncMsg.cause());
050                future.fail(asyncMsg.cause());
051            }
052        });
053        return future;
054    }
055
056    @Override
057    public Future<Optional<Ads>> get(@NonNull final Long id) {
058        final Future<Optional<Ads>> future = Future.future();
059        this.eventBus.<Ads>send(DatabaseVerticle.DB_AD_GET_BY_ID, id, asyncMsg -> {
060            if(asyncMsg.succeeded())
061                future.complete(Optional.ofNullable(asyncMsg.result().body()));
062            else {
063                log.error("Get error from bus resquest", asyncMsg.cause());
064                future.fail(asyncMsg.cause());
065            }
066        });
067        return future;
068    }
069
070    @Override
071    public Future<Void> delete(@NonNull final Long id) {
072        final Future<Void> future = Future.future();
073        this.eventBus.<Void>send(DatabaseVerticle.DB_AD_DELETE_BY_ID, id, asyncMsg -> {
074            if(asyncMsg.succeeded())
075                future.complete();
076            else {
077                log.error("Get error from bus resquest", asyncMsg.cause());
078                future.fail(asyncMsg.cause());
079            }
080        });
081        return future;
082    }
083
084    @Override
085    public Future<Long> createOrUpdate(@NonNull final Ad data) {
086        final Future<Long> future = Future.future();
087        this.eventBus.<Long>send(DatabaseVerticle.DB_AD_CREATE, DbConverter.toDB(data), asyncMsg -> {
088            if(asyncMsg.succeeded())
089                //future.complete(DbConverter.toAPI(asyncMsg.result().body()));
090                future.complete(asyncMsg.result().body());
091            else {
092                log.error("Get error from bus resquest", asyncMsg.cause());
093                future.fail(asyncMsg.cause());
094            }
095        });
096        return future;
097    }
098
099    void nextRotate(@NonNull final RoutingContext routingContext) {
100        final Optional<Long> id = Optional.ofNullable(((RequestParameters) routingContext.get("parsedParameters")).pathParameter("id").getLong());
101        this.get(rotate.getAndIncrement()).setHandler(res -> {
102            if(res.succeeded()) {
103                if(res.result().isPresent())
104                    routingContext.response().setStatusCode(200).end(Json.encode(res.result().get()));
105                else {
106                    //maybe the rotate arrive at max, try with reset
107                    rotate.set(0L);
108                    this.get(rotate.get()).setHandler(res2 -> {
109                        if(res2.succeeded() && res2.result().isPresent())
110                            routingContext.response().setStatusCode(200).end(Json.encode(res2.result().get()));
111                        else
112                            routingContext.response().setStatusCode(500).end(new JsonObject().put("error", new JsonObject().put("reason", "An error occur on the server")).toString());
113                    });
114                }
115            } else
116                routingContext.response().setStatusCode(500).end(new JsonObject().put("error", new JsonObject().put("reason", "An error occur on the server")).toString());
117        });
118    }
119}
