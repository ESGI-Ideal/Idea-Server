073        final Future<List<Images>> future = Future.future();
074        this.eventBus.<List<Images>>send(DatabaseVerticle.DB_IMAGE_GET_ALL, null, asyncMsg -> {
075            if(asyncMsg.succeeded())
076                future.complete(asyncMsg.result().body());
077            else {
078                log.error("Get error from bus resquest", asyncMsg.cause());
079                future.fail(asyncMsg.cause());
080            }
081        });
082        return future;
083    }

113    public void getFile(@NonNull final RoutingContext routingContext) {
114        final long id = ((RequestParameters) routingContext.get("parsedParameters")).pathParameter("id").getLong();
115        final String path = ((LocalStorage)this.storage).generatePath(TypeObject.Image, id);
116        this.vertx.fileSystem().exists(path, res -> {
117            if(res.succeeded())
118                routingContext.response().putHeader("Content-Type", "image/jpg").sendFile(res.result() ? path : this.noimg.toString());
119            else
120                RouteUtils.error(routingContext, res.cause().toString());
121        });
122    }

173    void coreupload(@NonNull final RoutingContext routingContext) {
174        final HttpServerRequest req = routingContext.request().pause();
175        String filename = UUID.randomUUID() + ".uploaded";
176        vertx.fileSystem().open(filename, new OpenOptions(), ares -> {
177            final AsyncFile file = ares.result();
178            final Pump pump = Pump.pump(req, file);
179            req.endHandler(v1 -> file.close(v2 -> {
180                System.out.println("Uploaded to " + filename);
181                req.response().end();
182            }));
183            pump.start();
184            req.resume();
185        });
186    }
