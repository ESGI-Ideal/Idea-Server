001package fr.esgi.ideal.api.database.codec;
002
003import io.vertx.core.buffer.Buffer;
004import io.vertx.core.eventbus.MessageCodec;
005import io.vertx.core.json.JsonObject;
006
007abstract public class AbstractMessageCodec<T> implements MessageCodec<T, T> {
008    abstract JsonObject encodeToWire(final T s);
009
010    /**
011     * Called by Vert.x when marshalling a message to the wire.
012     *
013     * @param buffer the message should be written into this buffer
014     * @param s      the message that is being sent
015     */
016    @Override
017    public void encodeToWire(final Buffer buffer, final T s) {
018        this.encodeToWire(s).writeToBuffer(buffer);
019    }
020
021    abstract T decodeFromWire(final JsonObject json);
022
023    /**
024     * Called by Vert.x when a message is decoded from the wire.
025     *
026     * @param pos    the position in the buffer where the message should be read from.
027     * @param buffer the buffer to read the message from
028     * @return the read message
029     */
030    @Override
031    public T decodeFromWire(final int pos, final Buffer buffer) {
032        final JsonObject json = new JsonObject();
033        json.readFromBuffer(pos, buffer);
034        return this.decodeFromWire(json);
035    }
036
037    /**
038     * If a message is sent <i>locally</i> across the event bus, this method is called to transform the message from
039     * the sent type S to the received type R
040     *
041     * @param s the sent message
042     * @return the transformed message
043     */
044    @Override
045    public T transform(final T s) {
046        return s;
047    }
048
049    /**
050     * The codec name. Each codec must have a unique name. This is used to identify a codec when sending a message and
051     * for unregistering codecs.
052     *
053     * @return the name
054     */
055    @Override
056    public String name() {
057        return this.getClass().getSimpleName();
058    }
059
060    /**
061     * Used to identify system codecs. Should always return -1 for a user codec.
062     *
063     * @return -1 for a user codec.
064     */
065    @Override
066    public byte systemCodecID() {
067        return -1;
068    }
069}
