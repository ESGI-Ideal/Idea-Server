048    @Override
049    public void start(@NonNull final Future<Void> startFuture) throws Exception {
050        log.debug("Starting verticle ...");
051        log.debug("config() = {}", this.config().encodePrettily());
052        this.start();
053        log.debug("Thumbnailator supported output formats : {}", String.join(", ", ThumbnailatorUtils.getSupportedOutputFormats()));
054        this.init_storage()
055            .compose(this::start_getOpenApiCotroller)
056            .compose(this::start_server)
057            .compose(x -> {
058                this.httpServer = x;
059                return Future.<Void>succeededFuture();
060            })
061            //.mapEmpty()
062            //.map(this::init_storage)
063            .setHandler(startFuture.completer());
064        log.debug("Starting complete");
065    }

072        fSrv.compose(v->{final Future<Void> next = Future.future(); this.storage.close(next); return next;})

075        this.storage.close(fStr);

115    private Future<Void> init_storage() {
116        final JsonObject conf = this.config().getJsonObject("storage");
117        try {
118            final StorageType type = StorageType.valueOf(conf.getString("type"));
119            this.storage = type.getNewInstance().get();
120            final Future<Void> future = Future.future();
121            this.storage.init(this.vertx, conf.getJsonObject("params", new JsonObject()), future);
122            return future;
123        } catch (final RuntimeException r) {
124            return Future.failedFuture(r);
125        }
126    }

278    private final static Function<RoutingContext, HttpServerResponse> respJson = ctx -> ctx.response().putHeader("content-type", "application/json; charset=utf-8");
279    private final static BiConsumer<RoutingContext, Object> outJson = (ctx, obj) -> respJson.apply(ctx).end(Json.encodePrettily(obj));
280
281    private static void ifId(@NonNull final RoutingContext ctx, @NonNull final BiConsumer<HttpServerResponse, Long> ifPresent) {
282        final String id = ctx.request().getParam("id");
283        if(id != null)
284            ifPresent.accept(respJson.apply(ctx), Long.valueOf(id));
285        else
286            ctx.response().setStatusCode(400).end();
287    }

303    private static Route addFaviconHandler(@NonNull final Route route) {
304        try {
305            route.handler(FaviconHandler.create(FSIO.getResourceAsExternal("favicon.ico").toString()));
306        } catch(final IOException e) {
307            log.warn("can't attach favicon", e);
308        }
309        return route;
