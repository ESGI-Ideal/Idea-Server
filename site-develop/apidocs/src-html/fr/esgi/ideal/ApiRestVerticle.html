050    @Override
051    public void start(@NonNull final Future<Void> startFuture) throws Exception {
052        log.debug("Starting verticle ...");
053        log.debug("config() = {}", this.config().encodePrettily());
054        this.start();
055        this.assets = Paths.get(this.config().getString("assets_dir", "./ideal-assets"));
056        if(Files.notExists(this.assets))
057            Files.createDirectory(this.assets.toAbsolutePath());
058        else
059            if(!Files.isDirectory(this.assets)) throw new InvalidPathException(this.assets.toString(), "The assets path is not an directory");
060        if(!Files.isReadable(this.assets) || !Files.isWritable(this.assets)) throw new InvalidPathException(this.assets.toString(), "Haven't right acces to assets directory");
061        log.debug("Thumbnailator supported output formats : {}", String.join(", ", ThumbnailatorUtils.getSupportedOutputFormats()));
062        this.start_getOpenApiCotroller()
063            .compose(this::start_server)
064            .compose(x -> {
065                this.httpServer = x;
066                return Future.<Void>succeededFuture();
067            })
068            .setHandler(startFuture);
069        log.debug("Starting complete");
070    }
071
072    @Override
073    public void stop(@NonNull final Future<Void> stopFuture) throws Exception {
074        log.debug("Stopping verticle");
075        this.httpServer.close(stopFuture.completer());
076    }

259    private final static Function<RoutingContext, HttpServerResponse> respJson = ctx -> ctx.response().putHeader("content-type", "application/json; charset=utf-8");
260    private final static BiConsumer<RoutingContext, Object> outJson = (ctx, obj) -> respJson.apply(ctx).end(Json.encodePrettily(obj));
261
262    private static void ifId(@NonNull final RoutingContext ctx, @NonNull final BiConsumer<HttpServerResponse, Long> ifPresent) {
263        final String id = ctx.request().getParam("id");
264        if(id != null)
265            ifPresent.accept(respJson.apply(ctx), Long.valueOf(id));
266        else
267            ctx.response().setStatusCode(400).end();
268    }

284    private static Route addFaviconHandler(@NonNull final Route route) {
285        try {
286            route.handler(FaviconHandler.create(FSIO.getResourceAsExternal("favicon.ico").toString()));
287        } catch(final IOException e) {
288            log.warn("can't attach favicon", e);
289        }
290        return route;
