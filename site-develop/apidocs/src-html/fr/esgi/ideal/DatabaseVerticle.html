102    @Override
103    public void start(final Future<Void> startFuture) throws Exception {
104        log.debug("Starting verticle ...");
105        log.debug("config() = {}", this.config().encodePrettily());
106        this.start();
107        this.vertx.eventBus().registerDefaultCodec(Users.class, new UsersMessageCodec());
108        this.vertx.eventBus().registerDefaultCodec(Ads.class, new AdsMessageCodec());
109        this.vertx.eventBus().registerDefaultCodec(Articles.class, new ArticlesMessageCodec());
110        this.vertx.eventBus().registerDefaultCodec(Partners.class, new PartnersMessageCodec());
111        this.vertx.eventBus().registerDefaultCodec(Images.class, new ImagesMessageCodec());
112        this.vertx.eventBus().registerCodec(new UsersListMessageCodec());
113        this.vertx.eventBus().registerCodec(new AdsListMessageCodec());
114        this.vertx.eventBus().registerCodec(new ArticlesListMessageCodec());
115        this.vertx.eventBus().registerCodec(new PartnersListMessageCodec());
116        this.vertx.eventBus().registerCodec(new ImagesListMessageCodec());
117        /*.put("driver_class", "org.postgresql.Driver")
118        .put("url", "jdbc:postgresql://localhost:5432/ideal?tcpKeepAlive=true&loglevel=debug")
119        .put("password", "ideal-pwd")*/
120        final JsonObject ds = this.config().getJsonObject("datasource");
121        Optional.ofNullable(ds)
122                .ifPresent(obj -> {
123                    Optional.ofNullable(ds.getString("url"))
124                            .ifPresent(url -> obj.put("url", url.replaceFirst("jdbc:", "jdbc:p6spy:")));
125                    Optional.ofNullable(ds.getString("driver_class"))
126                            .ifPresent(driver -> obj.put("driver_class", "com.p6spy.engine.spy.P6SpyDriver"));
127        });
128        try {
129            final String dialect = this.config().getString("dialect");
130            if(dialect == null)
131                throw new InvalidPropertiesFormatException("A SQL dialect must be specified");
132            final SqlParam param = SqlParam.valueOf(dialect);
133            jqDialect = param.jooqSqlDialect;
134            initP6spy(param.p6spyParams, this.vertx.getOrCreateContext().config().getString("name", "ideal-def"));
135        } catch(final RuntimeException e) {
136            throw new InvalidPropertiesFormatException(e);
137        }
138        this.client = JDBCClient.createShared(this.vertx, ds);
139        //startLiquibase.setHandler(startFuture.completer());
140        this.initLiquibase(this.config().getJsonObject("liquibase", new JsonObject().put("mustSafeToRunUpdate", true).put("forceDrop", false)),
141                           startFuture/*.completer()*/);
142        this.initBusConsummers();
143        log.debug("Starting complete");
144    }

214    private final static AtomicBoolean toInitP6spy = new AtomicBoolean(true);
215    private synchronized static void initP6spy(@NonNull final P6Param params, @NonNull final String name) throws InvalidPropertiesFormatException { //all calls are synchronous, so in series
216        if(toInitP6spy.compareAndSet(true, false)) { //only the first call will init env
217            try {
218                System.setProperty(SpyDotProperties.OPTIONS_FILE_PROPERTY, FSIO.getResourceAsExternal("spy.properties").toAbsolutePath().toString());
219                P6ModuleManager.getInstance().reload(); // make sure to reinit
220                // clean table plz (we need to make sure that all the configured factories will be re-loaded)
221                //new DefaultJdbcEventListenerFactory().clearCache();
222
223                Optional.ofNullable(P6SpyOptions.getActiveInstance()).ifPresent(instance -> {
224                    Optional.ofNullable(instance.getDriverNames())
225                            .ifPresent(drivers -> drivers.forEach(driver -> log.debug("P6Spy driver : {}", driver)));
226                    log.debug("P6Spy JMX : {}", instance.getJmx());
227                    log.debug("P6Spy JMX prefix = {}", instance.getJmxPrefix());
228                    log.debug("P6Spy reload properties : {}", instance.getReloadProperties());
229                    log.debug("P6Spy StackTrace : {}", instance.getStackTrace());
230                    log.debug("P6Spy AutoFlush : {}", instance.getAutoflush());
231
232                    instance.setDriverlist(params.driverJDBC);
233                    params.dateFormat.ifPresent(instance::setDateformat);
234                    instance.setJmxPrefix(name+"_");
235
236                    log.debug("P6Spy driver-list = {}", instance.getDriverlist());
237                    log.debug("P6Spy DB dialect date format : {}", instance.getDatabaseDialectDateFormat());
238                    log.debug("P6Spy real DataSource : {}", instance.getRealDataSource());
239                });
240                System.setProperty(P6Param.P6SpyEnvPrefix+P6SpyOptions.DRIVERLIST, params.driverJDBC);
241                params.dateFormat.ifPresent(dateformat ->System.setProperty(P6Param.P6SpyEnvPrefix+P6SpyOptions.DATEFORMAT, dateformat));
242                System.setProperty(P6Param.P6SpyEnvPrefix+P6SpyOptions.JMX_PREFIX, "name"+"_");
243                //params.URI_base
244            } catch (final IOException e) {
245                throw new RuntimeException(e); //during jvm init, so no logger and future.fail()
246            }
247                //driver_class
248        }
249        //P6Util.forName(driverName);
250        //log.debug("FRAMEWORK USING DRIVER == " + DriverManager.getDriver(url).getClass().getName() + " FOR URL " + url);
251    }

253    private /*static*/ Future<Void> startLiquibase = Future.future();
254    private final static AtomicBoolean toInit = new AtomicBoolean(true);
255    private synchronized void initLiquibase(@NonNull final JsonObject params, @NonNull final Future<Void> future) {
256        if(toInit.compareAndSet(true, false)) {
257            log.debug("Choose for init database");
258            startLiquibase.setHandler(future.completer());
259            this.client.getConnection(ar -> {
260                if(ar.succeeded()) {
261                    try(final SQLConnection connection = ar.result()) {
262                        final Liquibase liquibase = new Liquibase(DB_CHANGELOG, new ClassLoaderResourceAccessor(), new JdbcConnection(connection.unwrap()));
263                        log.trace("liquibase's logger used : {}", liquibase.getLog().getClass().getName());
264                        //~ Get State
265                        log.debug("Use ChangeLog '{}'", liquibase.getChangeLogFile());
266                        Stream.of(liquibase.listLocks()).forEach(lock -> log.debug("lock in database : {} the {} by {}", lock.getId(), lock.getLockGranted(), lock.getLockedBy()));
267                        liquibase.listUnrunChangeSets(null, null)
268                                .forEach(cs -> log.info("Changelog unrun : {} by {} in file {}", cs.getId(), cs.getAuthor(), cs.getFilePath()));
269                        if(liquibase.isSafeToRunUpdate())
270                            log.info("Liquibase safe to update");
271                        else {
272                            log.warn("Liquibase not safe to update");
273                            if(params.getBoolean("mustSafeToRunUpdate", true))
274                                throw new IllegalStateException("Database is not safe to update");
275                        }
276                        liquibase.update(new Contexts());
277                        //liquibase.validate(); //throws if not
278                        if(!liquibase.listUnrunChangeSets(null, null).isEmpty()) {
279                            log.warn("Failed to upgrade the database!");
280                            if(params.getBoolean("dropBeforeUpgrade", false)) {
281                                log.warn("Drop database and re-run upgrade");
282                                liquibase.dropAll();
283                                liquibase.update((Contexts) null);
284                                //no check because dropped
285                            } else
286                                throw new LiquibaseException("All changeSets have not been executed. Missing changeSets to liquibaseExecute !");
287                        } //else
288                        this.startLiquibase.complete();
289                    } catch(final LiquibaseException|IllegalStateException e) {
290                        log.error("Error while checking/updating the database", e);
291                        this.startLiquibase.fail(e);
292                    }
293                } else
294                    this.startLiquibase.fail(ar.cause());
295            });
296        } else {
297            log.debug("Wait database initialisation");
298            future.complete();
299            //startLiquibase.updateAndGet(prev -> Future.<Void>future().setHandler(completer));
300        }
301    }

307        if(jqConf == null)
308            jqConf = new Settings();

316    private <M> void execSql(@NonNull final Message<M> msg, @NonNull final Function<DSLContext, ?> getter) {
317        this.client.getConnection(result -> {
318            if (result.succeeded()) {
319                final Object tmp = getter.apply(this.mapDslContext(result.result().unwrap()));
320                result.result().close();
321                msg.reply(Json.encode(tmp));
322            } else
323                msg.fail(-1, result.cause().getMessage());
324        });
325    }

327    private <M> void execSqlRaw(@NonNull final Message<M> msg, @NonNull final Function<DSLContext, ?> getter) {
328        this.client.getConnection(result -> {

334                msg.fail(-1, result.cause().getMessage());

338    private <M> void execSqlCodec(@NonNull final Message<M> msg, @NonNull final Class<? extends AbstractListMessageCodec> codec, @NonNull final Function<DSLContext, List<?>> getter) {
339        this.client.getConnection(result -> {
340            if (result.succeeded()) {
341                final Object tmp = getter.apply(this.mapDslContext(result.result().unwrap()));
342                result.result().close();
343                DeliveryOptions ops = new DeliveryOptions(); ops.setCodecName(codec.getSimpleName());
344                msg.reply(tmp, ops);
345            } else
346                msg.fail(-1, result.cause().getMessage());
347        });
348    }
349
350    private <M> void execSqlNoReturn(@NonNull final Message<M> msg, @NonNull final Consumer<DSLContext> getter) {
351        this.client.getConnection(result -> {
352            if (result.succeeded()) {
353                getter.accept(this.mapDslContext(result.result().unwrap()));
354                result.result().close();
355                msg.reply(null);
356            } else
357                msg.fail(-1, result.cause().getMessage());
358        });
359    }
